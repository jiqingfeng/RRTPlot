 146def validate_params(valid_keys: list):
 147    """参数验证装饰器工厂函数
 148
 149    用于验证被装饰方法的输入参数，自动过滤非白名单参数
 150
 151    Args:
 152        valid_keys (list): 允许的参数名列表
 153
 154    Returns:
 155        function: 参数验证装饰器
 156
 157    Example:
 158        >>> @validate_params(['color', 'linewidth'])
 159        >>> def draw_line(...):
 160    """
 161
 162    def decorator(func):
 163        @wraps(func)
 164        def wrapper(self, *args, **kwargs):
 165            filtered = {k: v for k, v in kwargs.items() if k in valid_keys}
 166            return func(self, *args, **filtered)
 167
 168        return wrapper
 169
 170    return decorator

 297    def base_obstacle(self, vertices: np.ndarray, **kwargs) -> Polygon:
 298        """绘制基础多边形障碍物（无标签）
 299
 300        Args:
 301            vertices (np.ndarray): 多边形顶点坐标数组，形状为(N,2)

 305            Polygon: 创建的多边形Patch对象

 308            会自动闭合未闭合的多边形顶点

 320        obstacle = Polygon(vertices, closed=True, **params)
 321        self.ax.add_patch(obstacle)
 322        return obstacle

 406    def add_edge(
 407        self,
 408        start: Tuple[float, float],
 409        end: Tuple[float, float],
 410        label: Optional[str] = None,
 411        label_visible: bool = False,
 412        **kwargs,
 413    ) -> Tuple[Annotation, Optional[Annotation]]:
 414        """添加带自动编号的箭头连接
 415
 416        Args:
 417            start (Tuple[float, float]): 起点坐标 (x, y)
 418            end (Tuple[float, float]): 终点坐标 (x, y)
 419            label (Optional[str]): 自定义标签文本，默认自动生成序号
 420            label_visible (bool): 是否显示边标签，默认False
 421            **kwargs: 箭头样式参数，可覆盖：
 422                - color: 箭头颜色
 423                - linewidth: 线宽
 424                - arrowstyle: 箭头样式
 425                - mutation_scale: 箭头大小
 426
 427        Returns:
 428            Tuple[Annotation, Optional[Annotation]]:
 429                (箭头对象, 文本标签对象) 元组，当label_visible=False时第二个元素为None
 430
 431        Example:
 432            >>> arrow, label = plotter.add_edge((0,0), (5,5),
 433            >>>                                color='blue',
 434            >>>                                label_visible=True)
 435        """
 436        # 更新编号
 437        self._element_counters["edge"] += 1
 438        # 获取自动编号
 439        if label is None:
 440            label = str(self._element_counters["edge"])
 441
 442        # 绘制基础箭头
 443        arrow = self.base_arrow(start, end, **kwargs)
 444
 445        # 计算标签位置
 446        dx = end[0] - start[0]
 447        dy = end[1] - start[1]
 448        off_x, off_y = np.array([-dy, dx]) / np.linalg.norm([-dy, dx])
 449
 450        label_x = start[0] + dx * 0.5 + off_x * self.style.arrow["label_offset"]
 451        label_y = start[1] + dy * 0.5 + off_y * self.style.arrow["label_offset"]
 452
 453        # 添加标签
 454        text = None
 455        if label_visible:
 456            text = self.ax.text(label_x, label_y, label, **self.style.arrow["fontdict"])
 457        return arrow, text

 459    def add_labeled_obstacle(
 460        self,
 461        vertices: np.ndarray,
 462        label: Optional[str] = None,
 463        label_visible: bool = False,
 464        **kwargs,
 465    ) -> Tuple[Polygon, Optional[Annotation]]:
 466        """添加带标签编号的障碍物多边形
 467
 468        Args:
 469            vertices (np.ndarray): 障碍物顶点坐标数组，形状为(N,2)
 470            label (Optional[str]): 自定义标签文本，默认自动编号
 471            label_visible (bool): 是否显示标签，默认False
 472            **kwargs: 障碍物样式参数（继承自base_obstacle）
 473
 474        Returns:
 475            Tuple[Polygon, Optional[Annotation]]:
 476                障碍物多边形对象和文本标签对象（当label_visible为True时非空）
 477
 478        Raises:
 479            ValueError: 当顶点数小于3时可能抛出异常
 480
 481        Example:
 482            >>> vertices = np.array([[0,0], [1,0], [1,1]])
 483            >>> obstacle, label = plotter.add_labeled_obstacle(vertices)
 484        """
 485        # 更新障碍物计数器
 486        self._element_counters["obstacle"] += 1
 487        # 自动生成标签编号
 488        if label is None:
 489            label = f"{self._element_counters['obstacle']}"
 490
 491        # 绘制基础障碍物
 492        obstacle = self.base_obstacle(vertices, **kwargs)
 493
 494        # 添加标签
 495        text = None
 496        if label_visible:
 497            # 计算精确几何中心
 498            closed_vertices = (
 499                vertices
 500                if np.allclose(vertices[0], vertices[-1])
 501                else np.vstack([vertices, vertices[0]])
 502            )
 503            centroid = np.mean(closed_vertices[:-1], axis=0)  # 排除重复的闭合点
 504
 505            # 获取障碍物实际颜色（处理颜色继承逻辑）
 506            obstacle_color = obstacle.get_facecolor()[:3]  # 获取RGB值，忽略alpha
 507
 508            # 计算反色（最大对比度）
 509            inverse_color = tuple(1 - val for val in obstacle_color)
 510
 511            text = self.ax.text(
 512                centroid[0],
 513                centroid[1],
 514                label,
 515                color=inverse_color,
 516                **{
 517                    **self.style.obstacle["fontdict"],
 518                    "ha": "center",  # 水平居中
 519                    "va": "center",  # 垂直居中
 520                },
 521            )
 522        return obstacle, text

 527    def connect_nodes(
 528        self, start_node: Circle, end_node: Circle, **kwargs
 529    ) -> Tuple[Polygon, Optional[Annotation]]:
 530        """
 531        连接两个RRT节点并绘制箭头
 532
 533        Args:
 534            start_node (Circle): 起始节点对象
 535            end_node (Circle): 终止节点对象
 536            **kwargs: 箭头样式参数
 537
 538        Returns:
 539            Tuple[Annotation, Optional[Annotation]]: (箭头对象, 标签文本对象)
 540
 541        Examples:
 542            >>> node1 = plotter.add_circle_node((0,0))
 543            >>> node2 = plotter.add_circle_node((5,5))
 544            >>> arrow, text = plotter.connect_nodes(node1, node2)
 545        """
 546        # 计算连接路径
 547        angle = np.arctan2(
 548            end_node.center[1] - start_node.center[1],
 549            end_node.center[0] - start_node.center[0],
 550        )
 551
 552        start = (
 553            start_node.center[0] + start_node.radius * np.cos(angle),
 554            start_node.center[1] + start_node.radius * np.sin(angle),
 555        )
 556        end = (
 557            end_node.center[0] - end_node.radius * np.cos(angle),
 558            end_node.center[1] - end_node.radius * np.sin(angle),
 559        )
 560
 561        # 绘制箭头
 562        arrow, arrow_text = self.add_edge(start, end, **kwargs)
 563
 564        return arrow, arrow_text

 599            start[0] + length * np.cos(angle),
 600            start[1] + length * np.sin(angle),

 655    def inward_connection(
 656        self, target_node: Circle, direction_angle: float, length: float, **kwargs
 657    ) -> Annotation:
 658        """创建指向目标节点的外部连接边
 659
 660        Args:
 661            target_node (Circle): 目标节点对象
 662            direction_angle (float): 连接方向角度（单位：度）
 663            length (float): 连接边总长度（需大于节点半径）
 664            **kwargs: 箭头样式参数
 665
 666        Returns:
 667            Dict[str, Any]: 包含箭头要素的字典，包含以下键：
 668                - arrow: 箭头对象
 669                - arrow_text: 标签对象
 670                - start_point: 起点坐标
 671                - end_point: 终点坐标
 672
 673        Raises:
 674            ValueError: 当长度小于节点半径时抛出
 675
 676        Example:
 677            >>> target = plotter.add_circle_node((5,5))
 678            >>> conn = plotter.inward_connection(target, 30, 6)
 679        """
 680        # 计算起点位置（目标节点外延反方向）
 681        radius = target_node.radius
 682        angle_rad = np.radians(direction_angle)
 683
 684        if length < radius:
 685            raise ValueError("Length must be greater than the radius.")
 686
 687        # 计算箭头终点（目标节点边缘）
 688        end_point = (
 689            target_node.center[0] - radius * np.cos(angle_rad),
 690            target_node.center[1] - radius * np.sin(angle_rad),
 691        )
 692
 693        # 计算起点位置（考虑长度和半径偏移）
 694        start_point = (
 695            end_point[0] - (length - radius) * np.cos(angle_rad),
 696            end_point[1] - (length - radius) * np.sin(angle_rad),
 697        )
 698        # 绘制箭头
 699        arrow, arrow_text = self.add_edge(start_point, end_point, **kwargs)
 700
 701        return {
 702            "arrow": arrow,
 703            "arrow_text": arrow_text,
 704            "start_point": start_point,
 705            "end_point": end_point,
 706        }

 887    def _draw_classic_rrt(self, root_node, params, elements, start_angle, start_length):
 888        """经典RRT树结构绘制内部实现
 889
 890        Args:
 891            root_node: 根节点对象，需包含position和children属性
 892            params (dict): 渲染参数配置
 893            elements (dict): 元素存储字典
 894            start_angle (float): 初始角度（弧度）
 895            start_length (float): 初始长度
 896
 897        Implementation:
 898            使用广度优先搜索遍历树结构，自动处理：
 899            - 节点半径差异（根节点特殊样式）
 900            - 边标签可见性
 901            - 根节点方向连接
 902        """
 903        from collections import deque
 904
 905        queue = deque([root_node])
 906
 907        while queue:
 908            node = queue.popleft()
 909
 910            # 绘制当前节点
 911            node_style = {
 912                "color": params["node_color"],
 913                "radius": (
 914                    params["root_radius"]
 915                    if node.parent is None
 916                    else self.style.node["radius"]
 917                ),
 918            }
 919            circle, _ = self.add_circle_node(
 920                tuple(node.position), label_visible=params["node_label"], **node_style
 921            )
 922            elements["nodes"].append(circle)
 923            # 记录当前节点的元素
 924            node.element = circle
 925
 926            # 绘制到父节点的边
 927            if node.parent is not None:
 928                edge, _ = self.connect_nodes(
 929                    node.parent.element,
 930                    node.element,
 931                    label_visible=params["edge_label"],
 932                )
 933                elements["edges"].append(edge)
 934                node.edge_element = edge
 935            elif start_angle is not None:
 936                # 根节点存在接入方向
 937                dic = self.inward_connection(
 938                    node.element,
 939                    start_angle,
 940                    start_length,
 941                    label_visible=params["edge_label"],
 942                )
 943                elements["edges"].append(dic["arrow"])
 944                node.edge_element = dic["arrow"]
 945
 946            # 添加子节点到队列
 947            queue.extend(node.children)

1103    def calculate_segment_relations(
1104        self, edge1: Annotation, edge2: Annotation
1105    ) -> Tuple[Optional[Tuple[float, float]], Optional[Tuple[float, float]], float]:
1106        """计算线段空间关系（几何核心方法）
1107
1108        Algorithm:
1109            1. 计算无限长直线交点
1110            2. 计算符号距离判断相对位置
1111            3. 计算有向角度差
1112
1113        Returns:
1114            tuple: 包含三个元素的元组：
1115                0: 交点坐标 (x,y) 或 None
1116                1: (到线段1的符号距离, 到线段2的符号距离)
1117                2: 从线段1到线段2的旋转角度（弧度）
1118
1119        Note:
1120            符号距离说明：负值表示在起点延长线方向，正值表示在终点延长线方向
1121        """
1122        # 获取线段坐标
1123        line1 = np.array([edge1.xyann, edge1.xy])
1124        line2 = np.array([edge2.xyann, edge2.xy])
1125
1126        # 计算直线交点
1127        intersection = self._line_intersection(line1, line2)
1128
1129        # 计算距离参数
1130        dist_params = (None, None)
1131        if intersection:
1132            dist_params = (
1133                self._calculate_signed_distance(line1, intersection),
1134                self._calculate_signed_distance(line2, intersection),
1135            )
1136
1137        # 计算有向角度差
1138        angle_diff = self._calculate_angle_diff(
1139            self._get_direction_vector(line1), self._get_direction_vector(line2)
1140        )
1141
1142        return intersection, dist_params, angle_diff

1149        min_radius: float,

1152        """【内部方法】根据符号距离延长线段到交点
1153
1154        Args:
1155            original_line (np.ndarray): 原始线段坐标数组 [起点, 终点]

1291            设直线1参数方程：p + t*u

1298        Reference:
1299            https://stackoverflow.com/a/565282
1300        """
1301        # 解线性方程组 [p + t*u = q + s*v]
1302        p, u = line1[0], line1[1] - line1[0]
1303        q, v = line2[0], line2[1] - line2[0]
1304
1305        cross_uv = np.cross(u, v)
1306        if np.isclose(cross_uv, 0):
1307            return None  # 平行或重合
1308
1309        w = q - p
1310        t = np.cross(w, v) / cross_uv
1311        return tuple(p + t * u)

 779
 780        # 确定标注位置

 788            intersection,
 789            2 * arc_radius,

 793            color=style_params["color"],
 794            linewidth=style_params["linewidth"],

 801            label_text,

 803            horizontalalignment="center",
 804            verticalalignment="center",
 805            **style_params["fontdict"],
 806        )

 809            "arc": arc,

 889        """经典RRT树结构绘制内部实现
 890
 891        Args:
 892            root_node: 根节点对象，需包含position和children属性

1104    def calculate_segment_relations(
1105        self, edge1: Annotation, edge2: Annotation
1106    ) -> Tuple[Optional[Tuple[float, float]], Optional[Tuple[float, float]], float]:
1107        """计算线段空间关系（几何核心方法）
1108
1109        Algorithm:
1110            1. 计算无限长直线交点
1111            2. 计算符号距离判断相对位置
1112            3. 计算有向角度差
1113
1114        Returns:
1115            tuple: 包含三个元素的元组：
1116                0: 交点坐标 (x,y) 或 None
1117                1: (到线段1的符号距离, 到线段2的符号距离)
1118                2: 从线段1到线段2的旋转角度（弧度）
1119
1120        Note:
1121            符号距离说明：负值表示在起点延长线方向，正值表示在终点延长线方向
1122        """
1123        # 获取线段坐标
1124        line1 = np.array([edge1.xyann, edge1.xy])
1125        line2 = np.array([edge2.xyann, edge2.xy])
1126
1127        # 计算直线交点
1128        intersection = self._line_intersection(line1, line2)
1129
1130        # 计算距离参数
1131        dist_params = (None, None)
1132        if intersection:
1133            dist_params = (
1134                self._calculate_signed_distance(line1, intersection),
1135                self._calculate_signed_distance(line2, intersection),
1136            )
1137
1138        # 计算有向角度差
1139        angle_diff = self._calculate_angle_diff(
1140            self._get_direction_vector(line1), self._get_direction_vector(line2)
1141        )
1142
1143        return intersection, dist_params, angle_diff

1145    def _extend_to_intersection(
1146        self,
1147        original_line: np.ndarray,
1148        signed_dist: float,
1149        intersection: np.ndarray,
1150        min_radius: float,
1151        style: dict,
1152    ) -> Tuple[np.ndarray, Optional[Annotation]]:
1153        """【内部方法】根据符号距离延长线段到交点
1154
1155        Args:
1156            original_line (np.ndarray): 原始线段坐标数组 [起点, 终点]
1157            signed_dist (float): 符号距离（负数表示需要反向延长）
1158            intersection (np.ndarray): 理论交点坐标
1159            min_radius (float): 最小延长半径阈值
1160            style (dict): 延长线样式配置
1161
1162        Returns:
1163            Tuple[np.ndarray, Optional[Annotation]]:
1164                (延长后的线段坐标数组, 延长线图形对象)
1165
1166        Algorithm:
1167            1. 根据符号距离判断延长方向
1168            2. 计算需要延长的长度
1169            3. 绘制延长线并返回新线段
1170        """
1171        start, end = original_line[0], original_line[1]
1172        direction = end - start
1173
1174        # 情况1：交点在线段内部
1175        if signed_dist == 0:
1176            # 检查终点到交点的剩余长度
1177            remaining_length = np.linalg.norm(end - intersection)
1178            if remaining_length < min_radius:
1179                new_end = end + direction / np.linalg.norm(direction) * (
1180                    min_radius - remaining_length
1181                )
1182                arrow = self._draw_extension(end, new_end, style)
1183                return np.array([start, new_end]), arrow
1184            return original_line, None
1185
1186        # 情况2：需要反向延长（起点之前）
1187        if signed_dist < 0:
1188            extend_length = abs(signed_dist)
1189            new_start = start - direction / np.linalg.norm(direction) * extend_length
1190            arrow = self._draw_extension(start, new_start, style)
1191            return np.array([new_start, end]), arrow
1192
1193        # 情况3：需要正向延长（终点之后）
1194        extend_length = signed_dist + min_radius
1195        new_end = end + direction / np.linalg.norm(direction) * extend_length
1196        arrow = self._draw_extension(end, new_end, style)
1197        return np.array([start, new_end]), arrow

1225    def _calculate_label_position(
1226        self, center: Tuple[float, float], v1: np.ndarray, v2: np.ndarray, offset: float
1227    ) -> Tuple[float, float]:
1228        """【内部方法】计算角度标注标签位置
1229
1230        Args:
1231            center (Tuple[float, float]): 夹角顶点坐标
1232            v1 (np.ndarray): 第一条边方向向量
1233            v2 (np.ndarray): 第二条边方向向量
1234            offset (float): 偏移距离系数
1235
1236        Returns:
1237            Tuple[float, float]: 标签位置坐标 (x, y)
1238
1239        Math:
1240            标签位置 = 夹角平分线方向 * offset + 顶点坐标
1241        """
1242        # 计算夹角平分线方向
1243        angle1 = np.arctan2(v1[1], v1[0])
1244        angle2 = np.arctan2(v2[1], v2[0])
1245        bisect_angle = (angle1 + angle2) / 2
1246
1247        # 添加垂直于平分线的偏移
1248        dx = offset * np.cos(bisect_angle)
1249        dy = offset * np.sin(bisect_angle)
1250
1251        return (center[0] + dx, center[1] + dy)

1253    def _calculate_signed_distance(
1254        self, line: np.ndarray, point: Tuple[float, float]
1255    ) -> float:
1256        """【内部方法】计算点到线段的符号距离
1257
1258        Args:
1259            line (np.ndarray): 线段坐标数组 [[x0,y0], [x1,y1]]
1260            point (Tuple[float, float]): 目标点坐标
1261
1262        Returns:
1263            float: 符号距离值：
1264                - 负值：点在起点延长线方向
1265                - 0：点在线段内部
1266                - 正值：点在终点延长线方向
1267        """
1268        start, end = line[0], line[1]
1269        vec_line = end - start
1270        vec_point = np.array(point) - start
1271
1272        # 计算投影参数t
1273        t = np.dot(vec_point, vec_line) / np.dot(vec_line, vec_line)
1274
1275        if t < 0:
1276            # 起点之前的距离（返回负值）
1277            return -np.linalg.norm(vec_point)
1278        elif t > 1:
1279            # 终点之后的距离（返回正值）
1280            return np.linalg.norm(np.array(point) - end)
1281        else:
1282            # 在线段内部
1283            return 0.0

1290        Math:
1291            使用线性代数方法求解两直线交点：
1292            设直线1参数方程：p + t*u
1293            直线2参数方程：q + s*v
1294            解方程组得到t值
1295
1296        Returns:
1297            Optional[Tuple]: 交点坐标或None（当直线平行时）
1298
1299        Reference:
1300            https://stackoverflow.com/a/565282
1301        """
1302        # 解线性方程组 [p + t*u = q + s*v]
1303        p, u = line1[0], line1[1] - line1[0]
1304        q, v = line2[0], line2[1] - line2[0]
1305
1306        cross_uv = np.cross(u, v)
1307        if np.isclose(cross_uv, 0):
1308            return None  # 平行或重合
1309
1310        w = q - p
1311        t = np.cross(w, v) / cross_uv
1312        return tuple(p + t * u)

1330    def _calculate_angle_diff(self, v1: np.ndarray, v2: np.ndarray) -> float:

1340                - 负值表示顺时针旋转角度

1363            v2 (np.ndarray): 目标方向向量 [dx2, dy2]

1367                (start_angle, end_angle): 圆弧起止角度（弧度）

1369                is_acute: 是否为锐角（<90度）

1376            >>> v2 = np.array([0, 1])
1377            >>> angles, diff, acute = plotter._calculate_angle(v1, v2)
1378            # 返回 (0.0, 1.5708), 1.5708, True
1379        """
1380        # 计算原始角度差并规范化到(-π, π]
1381        angle_v1 = np.arctan2(v1[1], v1[0])
1382        angle_v2 = np.arctan2(v2[1], v2[0])
1383        angle_diff = (angle_v2 - angle_v1 + np.pi) % (2 * np.pi) - np.pi
1384
1385        start_angle = angle_v1
1386        end_angle = angle_v2
1387        is_acute = angle_diff > 0
1388
1389        # 圆弧的绘制规则为从start_angle逆时针到end_angle
1390        # 确保圆弧的绘制方向与角度差一致
1391        if angle_diff < 0:
1392            start_angle, end_angle = end_angle, start_angle
1393        return (start_angle, end_angle), angle_diff, is_acute

1395    def _get_label_offset(self, v1, v2):
